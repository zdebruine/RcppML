Factor Net Applications: Real-World NMF with Composable Graphs

This vignette demonstrates factor_net() on six real-world problems, each with domain-specific metrics, cross-validation, and publication-quality plots. For API reference and theory, see vignette("factor-graphs").

1. Part-Based Decomposition: Handwritten Digits

NMF discovers additive “parts” — strokes and loops that combine to form each digit. We factorize the 8 × 8 pixel digits dataset (scikit-learn digits, 1 797 images, 10 classes) and visualize both the basis images and convergence.

data("digits_full", package = "RcppML")
A_digits <- as(t(digits_full), "dgCMatrix")  # 64 pixels × 1797 images
cat(nrow(A_digits), "pixels ×", ncol(A_digits), "images\n")
#> 64 pixels × 1797 images

# Direct nmf() for loss_history (convergence tracking)
res_digits <- nmf(A_digits, k = 10, maxit = 100, tol = 1e-10, seed = 42)

Convergence

loss_df <- data.frame(
  iteration = seq_along(res_digits@misc$loss_history),
  loss = res_digits@misc$loss_history
)
ggplot(loss_df, aes(iteration, loss)) +
  geom_line(linewidth = 0.8, color = "#2166AC") +
  geom_point(size = 1, color = "#2166AC") +
  scale_y_log10(labels = scales::comma) +
  labs(x = "Iteration", y = "MSE (log scale)",
       title = "NMF convergence on handwritten digits (k = 10)")

Basis Images (Learned Parts)

Each column of W is a 64-element vector that reshapes into an 8 × 8 “basis image.” These parts correspond to recognizable stroke patterns.

W <- res_digits@w  # 64 × 10

# Build data frame for faceted heatmap
basis_df <- do.call(rbind, lapply(1:10, function(f) {
  img <- matrix(W[, f], 8, 8, byrow = FALSE)
  expand.grid(row = 1:8, col = 1:8) |>
    transform(value = as.vector(img), factor = paste("Part", f))
}))
basis_df$factor <- factor(basis_df$factor, levels = paste("Part", 1:10))

ggplot(basis_df, aes(col, row, fill = value)) +
  geom_tile() +
  scale_fill_viridis_c(option = "inferno") +
  scale_y_reverse() +
  facet_wrap(~ factor, nrow = 2) +
  coord_equal() +
  theme_void() +
  theme(strip.text = element_text(size = 9, margin = margin(b = 2)),
        legend.position = "none") +
  labs(title = "NMF basis images: learned stroke parts")

Cross-Validation: Optimal Rank

inp_d <- factor_input(A_digits, "digits")
cv_digits <- cross_validate_graph(
  inputs = inp_d,
  layer_fn = function(p) inp_d |> nmf_layer(k = p$k, L1 = p$L1),
  params = list(k = c(5L, 10L, 15L, 20L, 30L), L1 = c(0, 0.01, 0.1)),
  config = factor_config(maxit = 80, tol = 1e-5, seed = 42,
                         test_fraction = 0.15),
  reps = 3, seed = 42, verbose = FALSE
)

cv_df <- cv_digits$summary[!is.na(cv_digits$summary$mean_test_loss), ]
cv_df$L1_label <- factor(paste("L1 =", cv_df$L1))

ggplot(cv_df, aes(k, mean_test_loss, color = L1_label)) +
  geom_line(linewidth = 0.8) +
  geom_pointrange(aes(ymin = mean_test_loss - se_test_loss,
                      ymax = mean_test_loss + se_test_loss), size = 0.4) +
  scale_color_brewer(palette = "Set2") +
  labs(x = "Rank (k)", y = "Mean test loss ± SE",
       color = "L1 regularization",
       title = "CV rank selection for digits NMF") +
  theme(legend.position = c(0.7, 0.8))

Result: Rank 10 with light L1 regularization gives optimal generalization — matching the 10 digit classes.

2. Recommendation System: MovieLens Collaborative Filtering

For recommendation, only observed ratings matter — predicting zeros is meaningless. mask_zeros = TRUE evaluates on held-out observed ratings only, while mask_zeros = FALSE evaluates reconstruction of the full matrix (including the 97% zeros).

data("movielens", package = "RcppML")
A_ml <- movielens  # 3867 movies × 610 users
cat(nrow(A_ml), "movies ×", ncol(A_ml), "users\n")
#> 3867 movies × 610 users
cat("Sparsity:", round(1 - length(A_ml@x) / prod(dim(A_ml)), 3), "\n")
#> Sparsity: 0.968
cat("Ratings:", range(A_ml@x), "\n")
#> Ratings: 1 5

Standard vs. Recommendation Mode

modes <- expand.grid(k = c(5, 10, 20, 30), mask_zeros = c(FALSE, TRUE))
mode_results <- do.call(rbind, lapply(seq_len(nrow(modes)), function(i) {
  inp <- factor_input(A_ml, "movies")
  L1 <- inp |> nmf_layer(k = modes$k[i], name = "layer")
  net <- factor_net(inp, L1, config = factor_config(
    maxit = 100, tol = 1e-5, seed = 42,
    test_fraction = 0.1, mask_zeros = modes$mask_zeros[i]))
  res <- fit(net)
  lr <- res$layers[["layer"]]
  data.frame(k = modes$k[i],
             mode = ifelse(modes$mask_zeros[i], "Recommendation\n(observed only)",
                           "Standard\n(all entries)"),
             test_loss = lr$test_loss)
}))

ggplot(mode_results, aes(k, test_loss, color = mode)) +
  geom_line(linewidth = 0.8) +
  geom_point(size = 2.5) +
  scale_color_manual(values = c("#D95F02", "#1B9E77")) +
  labs(x = "Rank (k)", y = "Test loss (MSE)",
       color = NULL,
       title = "MovieLens: standard NMF vs. recommendation mode") +
  theme(legend.position = c(0.25, 0.85))

In standard mode, test loss rises with rank (overfitting to zeros). In recommendation mode (mask_zeros = TRUE), optimal rank is around 10 — fitting observed ratings without hallucinating structure in unrated entries.

3. Semi-Supervised Leukemia Classification (Golub)

The Golub dataset (38 patients × 5 000 genes) is a classic cancer classification benchmark. NMF naturally separates ALL from AML — we use guides to demonstrate semi-supervised learning with partial labels.

data("golub", package = "RcppML")
A_gol <- as(t(golub), "dgCMatrix")  # 5000 genes × 38 patients
labels_text <- sub("_.*", "", colnames(A_gol))
label_fac <- factor(labels_text)
label_int <- as.integer(label_fac) - 1L
cat(nrow(A_gol), "genes ×", ncol(A_gol), "patients\n")
#> 5000 genes × 38 patients
cat("Labels:", paste(levels(label_fac), table(label_fac), sep = " = ", collapse = ", "), "\n")
#> Labels: ALL = 27, AML = 11

Unsupervised Separation

Even at rank 2, NMF embeds ALL and AML into separable clusters:

res_gol <- nmf(A_gol, k = 2, seed = 42, maxit = 100)
emb_df <- data.frame(
  F1 = res_gol@h[1, ],
  F2 = res_gol@h[2, ],
  Cancer = label_fac
)

ggplot(emb_df, aes(F1, F2, color = Cancer, shape = Cancer)) +
  geom_point(size = 3) +
  scale_color_manual(values = c("ALL" = "#2166AC", "AML" = "#B2182B")) +
  labs(x = "NMF Factor 1", y = "NMF Factor 2",
       title = "Unsupervised NMF perfectly separates ALL from AML (k = 2)")

Guide Effect Across Lambda

lambdas <- c(0, 0.1, 0.5, 1, 5, 10)
set.seed(42)
test_idx <- sort(sample(38, 11))

sweep_gol <- do.call(rbind, lapply(lambdas, function(lam) {
  if (lam == 0) {
    m <- nmf(A_gol, k = 3, seed = 42, maxit = 200)
  } else {
    m <- nmf(A_gol, k = 3, seed = 42, maxit = 200,
             guides = guide_classifier(label_int, lambda = lam))
  }
  ev <- classify_embedding(t(m@h), label_int, test_idx = test_idx, k = 3)
  mse <- evaluate(m, A_gol)
  data.frame(lambda = lam, accuracy = ev$accuracy, mse = mse)
}))

4. Semi-Supervised Cell Typing: CITE-seq with Guide Classifier

CITE-seq measures both RNA and protein in the same cells, with expert annotations available for 27 fine-grained cell types (L2 labels). We set k = 27 — one factor per cell type — so the guide classifier can align each factor with a biological class.

library(SeuratData)
library(Seurat)

data("bmcite")
bmcite <- UpdateSeuratObject(bmcite)
rna <- bmcite[["RNA"]]$counts
ct <- bmcite$celltype.l2
cat(nrow(rna), "genes ×", ncol(rna), "cells\n")
#> 17009 genes × 30672 cells
cat(length(unique(ct)), "cell types (L2)\n")
#> 27 cell types (L2)

# Geometric sketch: random 3000 cells, top 2000 variable genes
set.seed(42)
idx <- sample(ncol(rna), 3000)
A_cite <- rna[, idx]
ct_sub <- ct[idx]
# Select top variable genes
rs <- Matrix::rowSums(A_cite)
rs2 <- Matrix::rowSums(A_cite^2)
n <- ncol(A_cite)
gene_var <- (rs2 / n) - (rs / n)^2
top_genes <- order(gene_var, decreasing = TRUE)[1:2000]
A_cite <- as(A_cite[top_genes, ], "dgCMatrix")
cat("Subsampled:", nrow(A_cite), "genes ×", ncol(A_cite), "cells\n")
#> Subsampled: 2000 genes × 3000 cells

ct_fac <- factor(ct_sub)
ct_int <- as.integer(ct_fac) - 1L

Unsupervised vs. Guided NMF

# Unsupervised
inp_cite <- factor_input(A_cite, "rna")
L_unsup <- inp_cite |> nmf_layer(k = 27, name = "layer")
net_unsup <- factor_net(inp_cite, L_unsup,
                        config = factor_config(maxit = 100, tol = 1e-5, seed = 42))
res_unsup <- fit(net_unsup)
H_unsup <- res_unsup$layers[["layer"]]$H

eval_unsup <- classify_embedding(t(H_unsup), ct_int, test_fraction = 0.2, k = 5, seed = 42)

# Guided (lambda = 0.1)
guide_cite <- guide_classifier(ct_int, lambda = 0.1, side = "H")
L_guided <- inp_cite |> nmf_layer(k = 27, name = "layer", H = H(guide = guide_cite))
net_guided <- factor_net(inp_cite, L_guided,
                         config = factor_config(maxit = 100, tol = 1e-5, seed = 42))
res_guided <- fit(net_guided)
H_guided <- res_guided$layers[["layer"]]$H

eval_guided <- classify_embedding(t(H_guided), ct_int, test_fraction = 0.2, k = 5, seed = 42)

cite_table <- data.frame(
  Model = c("Unsupervised NMF", "Guided NMF (λ = 0.1)"),
  Accuracy = c(eval_unsup$accuracy, eval_guided$accuracy),
  `Macro F1` = c(eval_unsup$macro_f1, eval_guided$macro_f1),
  check.names = FALSE
)
knitr::kable(cite_table, digits = 3,
             caption = "CITE-seq cell type classification (27 L2 types, k = 27)")

Semi-Supervised: Varying Label Fraction

fracs <- c(0.05, 0.1, 0.2, 0.5, 1.0)
semi_results <- do.call(rbind, lapply(fracs, function(frac) {
  set.seed(42)
  ss_labels <- ct_int
  if (frac < 1.0) {
    mask <- sample(c(TRUE, FALSE), length(ct_int), replace = TRUE,
                   prob = c(1 - frac, frac))
    ss_labels[mask] <- -1L
  }

  inp <- factor_input(A_cite, "rna")
  guide <- guide_classifier(ss_labels, lambda = 0.1, side = "H")
  L1 <- inp |> nmf_layer(k = 27, name = "layer", H = H(guide = guide))
  net <- factor_net(inp, L1, config = factor_config(maxit = 100, tol = 1e-5, seed = 42))
  res <- fit(net)
  ev <- classify_embedding(t(res$layers[["layer"]]$H), ct_int,
                           test_fraction = 0.2, k = 5, seed = 42)
  data.frame(fraction_labeled = frac, accuracy = ev$accuracy,
             n_labeled = sum(ss_labels >= 0))
}))

ggplot(semi_results, aes(fraction_labeled, accuracy)) +
  geom_line(linewidth = 0.8, color = "#D95F02") +
  geom_point(size = 2.5, color = "#D95F02") +
  geom_hline(yintercept = eval_unsup$accuracy, linetype = "dashed",
             color = "grey50") +
  annotate("text", x = 0.7, y = eval_unsup$accuracy + 0.005,
           label = "Unsupervised baseline", color = "grey50", size = 3.5) +
  scale_x_continuous(labels = scales::percent) +
  labs(x = "Fraction of cells labeled", y = "kNN accuracy (27 cell types)",
       title = "Semi-supervised NMF: accuracy vs. label budget")

Result: Even 5–10% labeled cells improve cell type classification over the unsupervised baseline, demonstrating effective label propagation through the NMF embedding. Setting k equal to the number of classes ensures each factor can specialize for one cell type.

5. Multi-Modal Integration: RNA + Protein

CITE-seq provides two complementary views of the same cells: RNA (broad transcriptomic coverage) and ADT (targeted protein markers). We use factor_shared() to discover a joint latent space, then compare classification accuracy across modalities.

# ADT (protein) modality from bmcite
adt <- bmcite[["ADT"]]$counts[, idx]
adt <- as(as.matrix(adt), "dgCMatrix")  # 25 proteins × 3000 cells
cat("RNA:", nrow(A_cite), "× ", ncol(A_cite), "\n")
#> RNA: 2000 ×  3000
cat("ADT:", nrow(adt), "×", ncol(adt), "\n")
#> ADT: 25 × 3000

Shared Factorization

factor_shared() forces RNA and ADT to share the same H matrix — a common embedding for all 3 000 cells that explains both gene expression and protein levels.

inp_rna <- factor_input(A_cite, "rna")
inp_adt <- factor_input(adt, "adt")
shared <- factor_shared(inp_rna, inp_adt)
L_shared <- shared |> nmf_layer(k = 27, name = "shared")
net_mm <- factor_net(list(inp_rna, inp_adt), L_shared,
                     config = factor_config(maxit = 100, tol = 1e-5, seed = 42))
res_mm <- fit(net_mm)
H_mm <- res_mm$layers[["shared"]]$H
eval_mm <- classify_embedding(t(H_mm), ct_int, test_fraction = 0.2, k = 5, seed = 42)

cat("RNA-only accuracy:", round(eval_unsup$accuracy, 3), "\n")
#> RNA-only accuracy: 0.638
cat("Multi-modal accuracy:", round(eval_mm$accuracy, 3), "\n")
#> Multi-modal accuracy: 0.667
